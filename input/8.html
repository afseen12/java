//8. Enter the long integer from user using Scanner class

class ReadLong{
	public static void main(String st[]){
		java.util.Scanner scanner = new java.util.Scanner(System.in);
	try{
		System.out.print("Enter the long interger value : ");
		long number = scanner.nextLong();
		System.out.println("You Entered : "+number);
	}
	catch(Throwable ex){
		System.out.println("Please enter the valid long integer value : "+ex.getMessage());
	}
	}
}

Output:

Enter the long interger value : 9273672890
You Entered : 9273672890
