//java method to count all word in a string

    class A
    {
        public static void main(String args[])
        {
            String str="miss";
            
            int wordCount=countWords(str);
            System.out.println("Word count:"+wordCount);
        }
    
            static int countWords(String str)
            {
                if(str==null)
                {
                    return 0;
                }
                return str.split("").length;
            }
    }

Output:

Word count:4
