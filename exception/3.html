//3. Write a java program to create  method that reads a file and throws an exception if the file is not found.

class ReadFile{
	public static void main(String st[]) 
	{
		try{
			readFile("file.txt");
		}
		catch(Throwable ex){
			System.out.println("File not found: "+ex.getMessage());
		}
	}
	static void readFile(String fileName) throws java.io.FileNotFoundException
	{
		java.io.File file = new java.io.File(fileName);
		java.util.Scanner scanner = new java.util.Scanner(file);
		while(scanner.hasNextLine()){
			System.out.println(scanner.nextLine());
		}
	}
}

Output:

File not found: file.txt (The system cannot find the file specified)
