/*11. Create a program, showing an example of super keyword.*/

    class Animal{
        void Cat(){
            System.out.println("Cat is eating");
        }
    }
    
    class Dog extends Cat {
        void Cat(){
            System.out.println("Cat is Domestic Animal");
        }
        void eat(){
            super.Cat();
        }
    }
    
    class Main{
        public static void main(String args[]){
            Dog obj = new Dog();
            obj.eat();
            obj. Cat();
        }
    }



Output:

Cat is eating
Cat is Domestic Animal
