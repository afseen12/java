/*14. Create a program, showing an example of dynamic method dispatch.*/

    class Parent{
        void Account(){
            System.out.println("Accounts should be handled manually");
        }
        void Worship(){
            System.out.println("Worship must be done early morning");
        }
    }
    
    class Child extends Parent{
        void Account(){
            System.out.println("Accounts should be handled through softwares");
        }
        void Store(){
            System.out.println("Grocery Store");
        }
    }
    
    class Main{
        public static void main(String args[]){
            Parent obj = new Child();
            obj.Account();
            obj.Worship();
            //obj.Store(); we can't call obj.store
        }
    }

Output:

Accounts should be handled through softwares
Worship must be done early morning
