/*12. Create a program, showing an example of super function*/

    class A{
    int x;
        A(int y){
            x = y;
        }
    
    }
    
    class B extends A{
        B(){
            super(10);
        }
    
    }
    
    class Main{
        public static void main(String args[]){
            B obj = new B();
            System.out.println("The value of x is "+obj.x);
        }
    }

Output:

The value of x is 10
