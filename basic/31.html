// Java Program to find area of rhombus

    class AreaOfRhombus{
        double diagonal1; 
        double diagonal2;
        AreaOfRhombus(double d1, double d2){
            diagonal1 = d1;
            diagonal2 = d2;
        }
        double areaOfRhombus(){
            return (diagonal1*diagonal2)/2;
        }
    }
    
    class Main{
        public static void main(String st[]){
        AreaOfRhombus rhombus = new AreaOfRhombus(3,6);
        System.out.println("Area of Rhombus : "+rhombus.areaOfRhombus());
        }
    }


Output:

Area of Rhombus : 9.0
