//Java Program to find volume of cone

    class VolumeOfCone{
        double height; 
        double radius;
        VolumeOfCone(double r, double h){
            radius = r;
            height = h;
        }
        double VolumeOfCone(){
            return (3.14)*(radius*radius)*(height/3);
        }
    }
    
    class Main{
        public static void main(String st[]){
        VolumeOfCone cone = new VolumeOfCone(3,4);
        System.out.println("Volume of cone : "+cone.VolumeOfCone());
        }
    }

Output:

Volume of cone : 37.68
