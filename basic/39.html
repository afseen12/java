//Java program to find surface area of cylinder

    class SurfaceAreaOfCylinder{
        double height;
        double radius;
        SurfaceAreaOfCylinder(double h, double r){
            height = h;
            radius = r;
        }
        double surfaceAreaOfCylinder(){
            return (2)*(3.14)*(radius)*(height + radius);
        }
    }
    
    class Main{
        public static void main(String st[]){
        SurfaceAreaOfCylinder cylinder = new SurfaceAreaOfCylinder(3,4);
        System.out.println("Surface area of cylinder : "+cylinder.surfaceAreaOfCylinder());
        }
    }

Output:

Surface area of cylinder : 175.84
