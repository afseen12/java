//Write a Java program to swap the first and last elements of an array and create a new array

    class SwapElement{
        public static void main(String st[]){
        int array[] = {1,2,3,4,5};
        System.out.print("Original Array : ");
        for(int i = 0; i<array.length; i++){
            System.out.print(array[i]+ " ");
        }
        int swapArray[] = new int[array.length];
        swapArray[0] = array[array.length-1];
        swapArray[array.length - 1] = array[0];
        
        for(int i = 1; i<array.length-1; i++){
            swapArray[i] = array[i];
        }
        System.out.println();
        System.out.print("Swapped Array : ");
        for(int i = 0; i<array.length; i++){
            System.out.print(swapArray[i]+ " ");
        }
    
    }}


Output:

Original Array : 1 2 3 4 5
Swapped Array : 5 2 3 4 1
