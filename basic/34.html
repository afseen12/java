//Java Program to find volume of sphere

    class VolumeOfSphere{
        double radius; 
        
        VolumeOfSphere(double r){
            radius = r;
        }
        double volumeOfSphere(){
            return (4/3.0)*(3.14)*(radius*radius*radius);
        }
    }
    
    class Main{
        public static void main(String st[]){
        VolumeOfSphere sphere = new VolumeOfSphere(3);
        System.out.println("Volume of sphere : "+sphere.volumeOfSphere());
        }
    }
