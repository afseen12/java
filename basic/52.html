// Write a Java program to take three numbers from the user and print the greatest number

    class GreatestNum {
        public static void main(String st[]) {
            java.util.Scanner scanner = new java.util.Scanner(System.in);
            System.out.print("Enter the first digit : ");
            int num1 = scanner.nextInt();
            System.out.print("Enter the second digit : ");
            int num2 = scanner.nextInt();
            System.out.print("Enter the third digit : ");
            int num3 = scanner.nextInt();
    
            int greatest = greatest(num1, num2, num3);
            System.out.println("The Greatest Number is : " + greatest);
        }
    
        static int greatest(int num1, int num2, int num3) {
            if (num1 >= num2 && num1 >= num3) {
                return num1;
            } else if (num2 >= num1 && num2 >= num3) {
                return num2;
            } else {
                return num3;
            }
        }
    }

Output:
