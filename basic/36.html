//Java Program to find volume of cuboid

    class VolumeOfCuboid{
        double length; 
        double breadth;
        double height;
        VolumeOfCuboid(double l, double h, double b){
            length = l; 
            breadth = b;
            height = h;
        }
        double volumeOfCuboid(){
            return length*breadth*height;
        }
    }
    
    class Main{
        public static void main(String st[]){
        VolumeOfCuboid cuboid = new VolumeOfCuboid(3,4,3);
        System.out.println("Volume of cuboid : "+cuboid.volumeOfCuboid());
        }
    }


Output:

Volume of cuboid : 36.0
