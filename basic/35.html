//Java Program to find volume of cylinder

    class VolumeOfCylinder{
        double radius; 
        double height;
        VolumeOfCylinder(double r, double h){
            radius = r;
            height = h;
        }
        double volumeOfCylinder(){
            return 3.14*radius*radius*height;
        }
    }
    
    class Main{
        public static void main(String st[]){
        VolumeOfCylinder cylinder = new VolumeOfCylinder(3,4);
        System.out.println("Volume of cylinder : "+cylinder.volumeOfCylinder());
        }
    }
