//Java Program to find area of parallelogram

    class AreaOfParallelogram{
        double base; 
        double height;
        AreaOfParallelogram(double b, double h){
            base = b;
            height = h;
        }
        double areaOfParallelogram(){
            return base*height;
        }
    }
    
    class Main{
        public static void main(String st[]){
        AreaOfParallelogram parallelogram = new AreaOfParallelogram(3,6);
        System.out.println("Area of parallelogram : "+parallelogram.areaOfParallelogram());
        }
    }


Output:

Area of parallelogram : 18.0
