//Java Program to find area of rectangle

    class AreaOfRectangle{
        double length; 
        double width;
        AreaOfRectangle(double l, double w){
            length = l;
            width = w;
        }
        double areaOfRectangle(){
            return length*width;
        }
    }
    
    class Main{
        public static void main(String st[]){
        AreaOfRectangle rectangle = new AreaOfRectangle(3,6);
        System.out.println("Area of rectangle : "+rectangle.areaOfRectangle());
        }
    }

Output:

Area of rectangle : 18.0
