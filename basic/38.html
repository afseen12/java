//Java program to find surface area of cuboid

    class SurfaceAreaOfCuboid{
        double height;
        double width; 
        double length;
        SurfaceAreaOfCuboid(double l, double w,double h){
            length = l;
            width = w;
            height = h;
        }
        double surfaceAreaOfCuboid(){
            return 2*(length*width+length*height+width*height);
        }
    }
    
    class Main{
        public static void main(String st[]){
        SurfaceAreaOfCuboid cuboid = new SurfaceAreaOfCuboid(3,4,3);
        System.out.println("Surface area of cuboid : "+cuboid.surfaceAreaOfCuboid());
        }
    }

Output:

Surface area of cuboid : 66.0
