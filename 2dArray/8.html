//8. Program to find the frequency of odd & even numbers in the given matrix

class EvenOddFrequency{
        public static void main(String st[]){
            int array[][] = {{1,2,3},{4,8,9},{6,5,5}};
            int odd = 0, even = 0;
    
            for(int i = 0; i<3; i++){
                for(int j = 0; j<3 ; j++){
                    if(array[i][j] % 2 == 0){
                        even++;
                    }
                    else{
                        odd++;
                    }
                }
            }
            System.out.println("Frequency of Odd Number in Matirx : "+odd);
            System.out.println("Frequency of Even Number in Matirx : "+even);
    
        }
    
    }

Output:

Frequency of Odd Number in Matirx : 5
Frequency of Even Number in Matirx : 4
