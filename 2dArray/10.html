//10. Program to find the transpose of a given matrix

class Transpose{
        public static void main(String st[]){
            int array[][] = {{2,3,4},{9,7,6},{5,7,3}};
            int transpose[][] = new int[array[0].length][array.length];
            for(int i = 0; i<3; i++){
                for(int j = 0; j<3; j++){
                    transpose[j][i] = array[i][j];
                }
            }
            System.out.println("Original Matrix:");
            printArray(array);
            System.out.println("Transpose Matrix:");
            printArray(transpose);
    
        }
        static void printArray(int array[][]){
            for(int i=0; i<3; i++){
                for(int j=0; j<3; j++){
                    System.out.print(array[i][j] + " ");
                }
                System.out.println();
            }
        }
    }

Output:

Original Matrix:
2 3 4
9 7 6
5 7 3
Transpose Matrix:
2 9 5
3 7 7
4 6 3
